<script type="text/javascript" >$(document).ready(function(){    $(".register input").each(function () {    //fix the reload problem	    if($(this).val() != "") {    	$(this).prev("label.inlined").hide();    }        // focus on the input element clicking on the label    $(this).prev("label.inlined").click(function(){    	$(this).next().focus();    })    $(this).focus(function () {    	if($(this).val() == "") {    		$(this).prev("label.inlined").fadeTo(200, 0.45);    	}    });    $(this).keypress(function () {    	$(this).prev("label.inlined").hide();    });    $(this).blur(function () {    	if($(this).val() == "") {    		$(this).prev("label.inlined").fadeTo(200, 1);    	}    });    });});   </script>
