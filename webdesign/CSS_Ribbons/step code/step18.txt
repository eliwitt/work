<script type="text/javascript" >$(document).ready(function(){    $(".register input").each(function () {    $(this).focus(function () {    	if($(this).val() == "") {    		$(this).prev("label.inlined").fadeTo(200, 0.45);    	}    });         $(this).keypress(function () {    	$(this).prev("label.inlined").hide();    });         $(this).blur(function () {    	if($(this).val() == "") {    		$(this).prev("label.inlined").fadeTo(200, 1);    	}    });    });});   </script>
